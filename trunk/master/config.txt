# config parameters used:
# ----------------------
#
C_CONFIG_ADM_TRACKER				tracker project to make svn commit under (e.g. "TASK")
C_CONFIG_ALIGNEDLIST				list of aligned distributive sets with IDs assigned (e.g. "dc.k1=1 dc.k2=2")
C_CONFIG_RELEASENUMBER				release number (e.g. 1.4.7)
C_CONFIG_APPVERSION				default major version (e.g. "1.0-SNAPSHOT")
C_CONFIG_APPVERSION_RELEASEFOLDER		release folder in svn (e.g. "R_1_2/prod-patch-1.1.4")
C_CONFIG_APPVERSION_SERVICES			default major version for wars (e.g. "2.6")
C_CONFIG_APPVERSION_TAG				default tag (e.g. "prod-2.6.5-candidate")
C_CONFIG_ARTEFACTDIR				local staging directory to download built items (e.g. "/build/artefacts/myproduct")
C_CONFIG_BRANCHNAME				default branch name (e.g. "prod-major")
C_CONFIG_BUILDER				type of builder ("maven"-default, or "gradle")
C_CONFIG_BUILDER_VERSION			use builder version (e.g. "2.1")
C_CONFIG_BUILDPATH				path to checkout/export source code to build (e.g. "/build/myproduct")
C_CONFIG_COMMIT_TRACKERLIST			set of tracker projects associated with product ("TASK SUPPORT")
C_CONFIG_DISTR_PATH				local path to store releases (e.g. "~/distr/myproject")
C_CONFIG_GITMIRRORPATH				local git repository location (e.g. "~/build/git")
C_CONFIG_JAVA_VERSION				use java version (e.g. "jdk1.6.0_29")
C_CONFIG_LAST_VERSION_BUILD			last minor tag (e.g. "~/myproduct/etc/last-prod-tag.txt")
C_CONFIG_MAVEN_ADDITIONAL_OPTIONS		additional maven build options (e.g. "-Dfile.encoding=UTF-8")
C_CONFIG_MAVEN_CFGFILE				maven configuration file (e.g. "~/.m2/settings.branch.xml")
C_CONFIG_MAVEN_PROFILES				maven -P profiles parameter (e.g. "all-components,all-modules")
C_CONFIG_MAVEN_VERSION				use maven version (e.g. "2.2.1")
C_CONFIG_MODULE_BUILD_OPTIONS_CORE		default build options for core (e.g. "-Mcs")
C_CONFIG_MODULE_BUILD_OPTIONS_WAR		default build options for war (e.g. "-Mwsn")
C_CONFIG_NEXT_MAJORRELEASE			next major release (e.g. 2.7)
C_CONFIG_NEXT_VERSION_BUILD			next minor version (e.g. "expr $LAST_PROD_TAG + 1")
C_CONFIG_NEXUS_BASE				base nexus path (e.g. "http://mynexus-hostport/nexus")
C_CONFIG_NEXUS_PATH_THIRDPARTY			nexus path for thirdpath download (e.g. http://mynexus-hostport/nexus/content/repositories/thirdparty")
C_CONFIG_NEXUS_REPO				repository to use (e.g. "builder-branch")
C_CONFIG_PROD_TAG				current prod tag (e.g. "prod-2.6.4")
C_CONFIG_PRODUCT				product name (e.g. "myproduct")
C_CONFIG_PRODUCT_DEPLOYMENT_HOME		fixed script root path (e.g. "~/myproduct")
C_CONFIG_REDISTPATH				staging area location (e.g. "/oracle/d/redist")
C_CONFIG_RELEASE_GROUPFOLDER			get release files from here (e.g. "R_1_2")
C_CONFIG_SCHEMAADMIN				admin database schema (e.g. "myadm")
C_CONFIG_SCHEMAADMIN_SCRIPTS			admin database schema table for scripts (e.g. "db_releases")
C_CONFIG_SCHEMAADMIN_RELEASES			admin database schema table for releases (e.g. "db_scripts")
C_CONFIG_SCHEMAALLLIST				database schema set for tnstype=all (e.g. "myadm appdata appint")
C_CONFIG_SCHEMAFEDLIST				database schema set for tnstype=fed (e.g. "myadm appint")
C_CONFIG_SCHEMAREGLIST				database schema set for tnstype=reg (e.g. "appdata")
C_CONFIG_SOURCE_CFG_LIVEROOTDIR	                full root svn path for live configuration saved (e.g. "http://mysvn-hostport/svn/releases/myproduct/configuration/live")
C_CONFIG_SOURCE_CFG_ROOTDIR			full root svn path for configuration (e.g. "http://mysvn-hostport/svn.releases/myproduct/configuration")
C_CONFIG_SOURCE_RELEASEROOTDIR			full root svn path for release (e.g. "http://mysvn-hostport/svn/releases/myproduct/changes")
C_CONFIG_SOURCE_SQL_GLOBALPENDING		full root svn path for pending scripts (e.g. "http://mysvn-hostport/svn/releases/myproduct/database")
C_CONFIG_SOURCE_SQL_LOGDIR			full path to SQL log directory (e.g. $C_CONFIG_PRODUCT_DEPLOYMENT_HOME/database/patches.log)
C_CONFIG_SVNNEW_AUTH				new sql auth (e.g. "--username=myuser --password=mypassword")
C_CONFIG_SVNNEW_PATH				new svn path (e.g. "http://mynewsvn-hostport/svn")
C_CONFIG_SVNOLD_AUTH				old sql auth (e.g. "--username=myuser --password=mypassword")
C_CONFIG_SVNOLD_PATH				old sql path (e.g. "http://myoldsvn-hostport/svn")
C_CONFIG_UPGRADEPATH				path to upgrade interface scripts (e.g. "/distr/common/upgrade")
C_CONFIG_USE_TEMPLATES				option to use templates (yes/no)
C_CONFIG_USE_WAR				option to use wars (yes/no)
C_CONFIG_VERSION_LAST_FULL			current prod full version (e.g. "1.4.5")
C_CONFIG_VERSION_BRANCH_MAJOR			first number of major version (e.g. "2")
C_CONFIG_VERSION_BRANCH_MINOR			second number of major version (e.g. "6")
C_CONFIG_VERSION_BRANCH_NEXTMINOR		next major release second number of major version (e.g. "7")
C_CONFIG_VERSION_NEXT_FULL			next prod full version (e.g. "1.4.6")
C_CONFIG_VERSIONBRANCH				current prod branch version (e.g. "1.3")
